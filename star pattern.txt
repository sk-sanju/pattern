#include <stdio.h>

int main()
{
    int i,j,n,s;
    printf("enter limit: ");
    scanf("%d",&n);
    for(i=0;i<=n;i++){
        for(s=0;s<=n-i;s++){
            printf(" ");
        }
        for(j=-n;j<=n;j++){
            if(j>-n+i&&j<n-i&&i!=n){
                printf("  ");
            }else{
            printf("* ");}
        }
        printf("\n");
    }
    return 0;
}
